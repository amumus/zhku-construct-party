1.获取某个用户每一项积分来源的积分总和，用于计算比例
SELECT score_id,SUM(integral_score) FROM
(
SELECT integral_score,score_id FROM user_score_detail WHERE user_id =1
) AS a
 GROUP BY score_id
2.获取某个用户每月的总分，uniapp用
 select a.`month`,sum(integral_score) as `score` from (SELECT * , MONTH(occur_time) as `month` FROM user_score_detail WHERE user_id =1)as a GROUP BY a.`month`
